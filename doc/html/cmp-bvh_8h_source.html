122 inline uint32_t LeftShift3(uint32_t x) {

124 if (x == (1 << 10))

144 static void RadixSort(std::vector<MortonPrimitive>* v) {
145 std::vector<MortonPrimitive> tempVector(v->size());
146 constexpr int bitsPerPass = 6;
147 constexpr int nBits = 30;
148 static_assert((nBits % bitsPerPass) == 0,
149 "Radix sort bitsPerPass must evenly divide nBits");
150 constexpr int nPasses = nBits / bitsPerPass;
151 for (int pass = 0; pass < nPasses; ++pass) {
152 // Perform one pass of radix sort, sorting _bitsPerPass_ bits
153 int lowBit = pass * bitsPerPass;
154 // Set in and out vector references for radix sort pass
155 std::vector<MortonPrimitive>& in = (pass & 1) ? tempVector : *v;
156 std::vector<MortonPrimitive>& out = (pass & 1) ? *v : tempVector;
157
158 // Count number of zero bits in array for current radix sort bit
159 constexpr int nBuckets = 1 << bitsPerPass;
160 int bucketCount[nBuckets] = { 0 };
161 constexpr int bitMask = (1 << bitsPerPass) - 1;
162 for (const MortonPrimitive& mp : in) {
163 int bucket = (mp.morton_code >> lowBit) & bitMask;
164 assert(bucket >= 0);
165 assert(bucket < nBuckets);
166 ++bucketCount[bucket];
167 }
168
169 // Compute starting index in output array for each bucket
170 int outIndex[nBuckets];
171 outIndex[0] = 0;
172 for (int i = 1; i < nBuckets; ++i)
173 outIndex[i] = outIndex[i - 1] + bucketCount[i - 1];
174
175 // Store sorted values in output array
176 for (const MortonPrimitive& mp : in) {
177 int bucket = (mp.morton_code >> lowBit) & bitMask;
178 out[outIndex[bucket]++] = mp;
179 }
180 }
181 // Copy final result from _tempVector_, if needed
182 if (nPasses & 1)
183 std::swap(*v, tempVector);
184 }
